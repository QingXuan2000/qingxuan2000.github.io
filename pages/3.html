def fibonacci(n):
    """返回斐波那契数列的前 n 个数"""
    a, b = 0, 1
    result = []
    for _ in range(n):
        result.append(a)
        a, b = b, a + b
    return result

print(fibonacci(10))

  • pymdownx.caret (插入)
